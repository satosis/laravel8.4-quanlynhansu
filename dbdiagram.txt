  nhanvien_id int(10)
  ngay_san_xuat date

  ghi_chu text
  nguoi_danh_gia_id int(10)
  created_at timestamp
  updated_at timestamp
  deleted_at timestamp
}

Table users {
  id integer [primary key]
  nhanvien_id int(10)
  email varchar(50)
  email_verified_at timestamp
  password varchar(255)
  role tinyint(4)
  remember_token varchar(100)
  created_at timestamp
  updated_at timestamp
  deleted_at timestamp
}

  diachi varchar(255)

  trangthai tinyint(1)

  updated_at timestamp
  deleted_at timestamp
}
